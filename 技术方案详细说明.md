#### 文件上传服务实现
```java
@Service
public class FileUploadService {
    private final HttpUtil httpUtil;
    private final Base64Util base64Util;
    private final ConfigService configService;
    
    public CompletableFuture<UploadResult> uploadImage(File imageFile, ProductInfo productInfo) {
        return CompletableFuture.supplyAsync(() -> {
            try {
                // 1. 读取图片文件并转换为Base64
                String base64Image = base64Util.encodeFile(imageFile);
                
                // 2. 构建上传请求
                UploadRequest request = UploadRequest.builder()
                    .filename(imageFile.getName())
                    .imageType(getImageType())
                    .base64Data(base64Image)
                    .productInfo(productInfo)
                    .build();
                
                // 3. 发送HTTP请求
                String response = httpUtil.post("/api/upload", request);
                
                // 4. 解析响应结果
                return parseUploadResponse(response);
                
            } catch (Exception e) {
                return UploadResult.failure(e.getMessage());
            }
        });
    }
    
    public void uploadBatch(List<File> imageFiles, List<ProductInfo> productInfos, 
                           ProgressCallback progressCallback) {
        // 批量上传实现，支持进度回调
        ExecutorService executor = Executors.newFixedThreadPool(5);
        AtomicInteger completed = new AtomicInteger(0);
        int total = imageFiles.size();
        
        for (int i = 0; i < imageFiles.size(); i++) {
            File imageFile = imageFiles.get(i);
            ProductInfo productInfo = productInfos.get(i);
            
            CompletableFuture<UploadResult> future = uploadImage(imageFile, productInfo);
            future.thenAccept(result -> {
                int currentCompleted = completed.incrementAndGet();
                Platform.runLater(() -> {
                    progressCallback.onProgress(currentCompleted, total, result);
                });
            });
        }
    }
}
```

```java
@Service
@Transactional
public class AuthService {
    
    @Autowired
    private PicUserMapper picUserMapper;
    
    @Autowired
    private PasswordEncoder passwordEncoder;
    
    public ApiResponse<LoginResponse> login(LoginRequest request) {
        try {
            // 1. 参数验证
            if (StringUtils.isEmpty(request.getUsername()) || 
                StringUtils.isEmpty(request.getPassword())) {
                return ApiResponse.error("用户名和密码不能为空");
            }
            
            // 2. 查询用户
            PicUser user = picUserMapper.findByUsername(request.getUsername());
            if (user == null) {
                return ApiResponse.error("用户不存在");
            }
            
            // 3. 验证密码
            if (!passwordEncoder.matches(request.getPassword(), user.getPassword())) {
                return ApiResponse.error("密码错误");
            }
            
            // 4. 生成登录响应
            LoginResponse response = LoginResponse.builder()
                .userId(user.getId())
                .username(user.getUsername())
                .nickname(user.getNickname())
                .loginTime(LocalDateTime.now())
                .build();
            
            return ApiResponse.success(response);
            
        } catch (Exception e) {
            log.error("登录失败", e);
            return ApiResponse.error("登录失败: " + e.getMessage());
        }
    }
}
```

### 4.1 Base64图片处理
```java
@Component
public class Base64Util {
    
    public String encodeFile(File file) throws IOException {
        byte[] fileContent = Files.readAllBytes(file.toPath());
        return Base64.getEncoder().encodeToString(fileContent);
    }
    
    public void decodeToFile(String base64Data, String filePath) throws IOException {
        byte[] decodedBytes = Base64.getDecoder().decode(base64Data);
        Files.write(Paths.get(filePath), decodedBytes);
    }
    
    public boolean isValidBase64Image(String base64Data) {
        try {
            byte[] decodedBytes = Base64.getDecoder().decode(base64Data);
            // 检查文件头，判断是否为有效的图片格式
            return isValidImageFormat(decodedBytes);
        } catch (Exception e) {
            return false;
        }
    }
    
    private boolean isValidImageFormat(byte[] bytes) {
        if (bytes.length < 4) return false;
        
        // JPEG文件头: FF D8 FF
        if (bytes[0] == (byte)0xFF && bytes[1] == (byte)0xD8 && bytes[2] == (byte)0xFF) {
            return true;
        }
        
        // PNG文件头: 89 50 4E 47
        if (bytes[0] == (byte)0x89 && bytes[1] == 0x50 && 
            bytes[2] == 0x4E && bytes[3] == 0x47) {
            return true;
        }
        
        return false;
    }
}
```

### 4.2 Excel处理实现
```java
@Service
public class ExcelService {
    
    public List<ProductInfo> parseExcelFile(File excelFile) throws IOException {
        List<ProductInfo> productInfoList = new ArrayList<>();
        
        try (Workbook workbook = WorkbookFactory.create(excelFile)) {
            Sheet sheet = workbook.getSheetAt(0);
            
            // 跳过标题行，从第二行开始读取数据
            for (int rowIndex = 1; rowIndex <= sheet.getLastRowNum(); rowIndex++) {
                Row row = sheet.getRow(rowIndex);
                if (row == null) continue;
                
                ProductInfo productInfo = new ProductInfo();
                productInfo.setField01(getCellValue(row.getCell(0)));  // 商品编号
                productInfo.setField02(getCellValue(row.getCell(1)));  // 分类
                productInfo.setField03(getCellValue(row.getCell(2)));  // 订单号
                productInfo.setField04(getCellValue(row.getCell(3)));  // 面料
                productInfo.setField05(getCellValue(row.getCell(4)));  // 客户名
                productInfo.setField06(getCellValue(row.getCell(5)));  // 市场
                productInfo.setField07(getCellValue(row.getCell(6)));  // 套数
                productInfo.setField08(getCellValue(row.getCell(7)));  // SO日期
                productInfo.setField09(getCellValue(row.getCell(8)));  // 大货日期
                productInfo.setField10(getCellValue(row.getCell(9)));  // 工厂编号
                productInfo.setField11(getCellValue(row.getCell(10))); // 工厂
                productInfo.setField12(getCellValue(row.getCell(11))); // 描稿公司
                productInfo.setField13(getCellValue(row.getCell(12))); // 描稿人员
                productInfo.setField14(getCellValue(row.getCell(13))); // 备注
                productInfo.setField15(getCellValue(row.getCell(14))); // 理单员
                productInfo.setField16(getCellValue(row.getCell(15))); // 排序
                
                productInfoList.add(productInfo);
            }
        }
        
        return productInfoList;
    }
    
    private String getCellValue(Cell cell) {
        if (cell == null) return "";
        
        switch (cell.getCellType()) {
            case STRING:
                return cell.getStringCellValue().trim();
            case NUMERIC:
                if (DateUtil.isCellDateFormatted(cell)) {
                    return new SimpleDateFormat("yyyy-MM-dd").format(cell.getDateCellValue());
                } else {
                    return String.valueOf((long) cell.getNumericCellValue());
                }
            case BOOLEAN:
                return String.valueOf(cell.getBooleanCellValue());
            case FORMULA:
                return cell.getCellFormula();
            default:
                return "";
        }
    }
}
```
